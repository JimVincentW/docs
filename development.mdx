<Info>
  **Prerequisite** You should have installed docker along with docker compose.
</Info>





<Tabs>
  <Tab title="Open Politics">
    <Steps>
      <Step title="Clone the repository">
        <CodeGroup>
        ```bash bash
        git clone https://github.com/open-Politics/open-Politics.git
        ```
        </CodeGroup>
      </Step>
      <Step title="Change .env.example to .env">
        <CodeGroup>
        ```bash bash
        cd open-Politics
        mv .env.example .env
        ```

        <Icon icon="check" iconType="solid" /> The documentation you want, effortlessly

        </CodeGroup>
      </Step>
      <Step title="Run the docker compose file">
        <CodeGroup>
        ```bash bash
        docker compose up
        ```
        </CodeGroup>
      </Step>
      <Step title="Go to the dashboard">
        <Card title="Dashboard URL" img="/images/ssare_hub.png" color="green">
          ```bash bash 
          http://localhost:3000/desk_synthese
          ```
          <Frame>
          <img src="/images/opp_desk.png" />
          </Frame>
        </Card>
      </Step>
    </Steps>
  </Tab>
  <Tab title="SSARE">
    <Steps>
      <Step title="Clone the repository">
        <CodeGroup>
        ```bash bash
        git clone https://github.com/open-politics/SSARE.git
        ```
        </CodeGroup>
      </Step>
      <Step title="Change .env.example to .env">
        <CodeGroup>
        ```bash bash
        cd SSARE
        mv .env.example .env
        ```

      </Step>
      <Step title="Run the boot script">
        <CodeGroup>
        ```bash bash
        sudo bash boot.sh
        ```
        </CodeGroup>
      </Step>
      <Step title="Go to the app">
        <Card title="Dashboard URL" color="green">
          ```bash bash 
          http://localhost:3000/desk_synthese
          ```
          <Frame>
          <img src="/images/ssare_hub.png" />
          </Frame>
        </Card>
      </Step>
    </Steps>
  </Tab>
</Tabs>



<AccordionGroup>
  <Accordion title="Ports unavailable">
    Check your .env file to ensure the ports are correctly configured and not in use by other services.
  </Accordion>
  <Accordion title="At reserve">
    Tear down the current setup, clone the repository again, and copy the .env.example file to .env.
  </Accordion>
</AccordionGroup>
